‘ 1.create sheet
‘
Sub createSheet()
Dim i As Integer, blnFound As Boolean
blnFound = False
With ThisWorkbook
For i = 1 To .Sheets.Count
If .Sheets(i).Name = “operatorData” Then
blnFound = True
Exit For
End If
Next i

    If blnFound = False Then
        .Sheets.Add
        With ActiveSheet
            .Name = "operatorData"
        End With
    End If
End With

End Sub

‘
‘ create title

Sub createTitle()
Worksheets(“operatorData”).Cells(“1”, “A”) = “operatorId”
Worksheets(“operatorData”).Cells(“1”, “B”) = “分期金额总和”
Worksheets(“operatorData”).Cells(“1”, “C”) = “分期手续费总和”
Worksheets(“operatorData”).Cells(“1”, “D”) = “分期期数的平均数”
Worksheets(“operatorData”).Cells(“1”, “E”) = “对应分期期数下客户数总和”
Worksheets(“operatorData”).Cells(“1”, “F”) = “对应期的用户数”
End Sub

createSheet
